This means: to make local files work end-to-end through the existing clip
pipeline WITHOUT changing the clip router, we need a synthetic `url` value
that, when fed through `url_cache_key`, hashes to the same 16-char stem the
upload created. Strategy: store the file at `cache_dir/{content_hash[:16]}.mp4`
AND register an opaque pseudo-URL of the form `local:{content_hash[:16]}` in
url_registry.json so that `url_cache_key("local:{hash}") == content_hash[:16]`.
We achieve this by skipping `url_cache_key` for the local case and storing the
final 16-char stem directly in the registry under whatever pseudo-URL we
choose. Concretely: the registry already maps cache_key → URL (free-form
strings); the URL doesn't have to round-trip through url_cache_key for the
upload path — but DOES have to round-trip for `cache_status_for_url` and for
`POST /api/instructional`. Therefore the synthetic URL we expose to the client
MUST be a string whose sha256[:16] equals the file's content_hash[:16]. The
simplest reliable scheme: synthetic_url = f"local:{content_hash_full_64hex}"
and `key = url_cache_key(synthetic_url)` — both ends derive the SAME 16-char
stem from the SAME synthetic_url, with no need for the stem to equal anything
in particular.

Concrete identity scheme used in this plan:
- content_hash = sha256(file_bytes).hexdigest()  (full 64-hex)
- synthetic_url = f"local:{content_hash}"
- cache_key = url_cache_key(synthetic_url)  # 16 hex of sha256(synthetic_url)
- Stored at: cache_dir / f"{cache_key}.mp4"
- Registry: {cache_key: synthetic_url}
- This is identical in shape to URL-keyed entries. `cache_status_for_url`,
  preview, delete, list all work without modification.

```python
def ingest_local_video(
    cache_dir: Path,
    file_bytes: bytes,
    *,
    max_bytes: int,
) -> dict[str, Any]:
    """
    Persist user-supplied .mp4 bytes into the cache, identified by content hash.

    Identity scheme: synthetic_url = f"local:{sha256_hex}", cache_key derived
    via the same `url_cache_key` used for real URLs so the rest of the cache
    pipeline (status, preview, delete, list) treats the entry uniformly.
    """
    if not file_bytes:
        raise ValueError("Empty upload")
    if max_bytes > 0 and len(file_bytes) > max_bytes:
        raise ValueError(f"Upload exceeds {max_bytes} bytes limit")
    content_hash = hashlib.sha256(file_bytes).hexdigest()
    synthetic_url = f"local:{content_hash}"
    cache_key = url_cache_key(synthetic_url)
    cache_dir.mkdir(parents=True, exist_ok=True)
    out_path = cache_dir / f"{cache_key}.mp4"
    out_path.write_bytes(file_bytes)
    save_url_registry_entry(cache_dir, synthetic_url)
    return {
        "cache_key": cache_key,
        "synthetic_url": synthetic_url,
        "size_bytes": out_path.stat().st_size,
        "path": out_path,
    }
```

```python
@router.post("/cache/upload", response_model=LocalIngestResponse)
async def upload_local_video(file: UploadFile = File(...)) -> LocalIngestResponse:
    """Ingest a user-supplied .mp4 into the cache (no yt-dlp). Identity = content hash."""
    # Filename / extension guard. The upload's `filename` is attacker-controlled;
    # we use it ONLY to check the extension, never as a path.
    name = (file.filename or "").strip()
    if not name.lower().endswith(".mp4"):
        raise HTTPException(status_code=400, detail="Only .mp4 uploads are accepted.")

    # Content-type guard (best-effort; browsers usually set video/mp4).
    ctype = (file.content_type or "").lower()
    if ctype and not (ctype.startswith("video/") or ctype == "application/octet-stream"):
        raise HTTPException(status_code=400, detail=f"Unexpected content type: {ctype}")

    # Read in full. UploadFile is a SpooledTemporaryFile — large files spill to disk.
    data = await file.read()
    if not data:
        raise HTTPException(status_code=400, detail="Empty upload.")
    if len(data) > LOCAL_UPLOAD_MAX_BYTES:
        raise HTTPException(
            status_code=413,
            detail=f"Upload exceeds {LOCAL_UPLOAD_MAX_BYTES} bytes.",
        )

    # MP4 magic-byte sanity check: ftyp box at offset 4 (RFC 14496-12). Reject
    # anything that does not look like a real ISO/MP4 container — prevents the
    # cache directory from being filled with arbitrary user-supplied bytes.
    if len(data) < 12 or data[4:8] != b"ftyp":
        raise HTTPException(status_code=400, detail="File is not a valid MP4 (ftyp box missing).")

    settings = load_settings()
    try:
        result = ingest_local_video(
            settings.cache_dir,
            data,
            max_bytes=LOCAL_UPLOAD_MAX_BYTES,
        )
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e)) from e
    return LocalIngestResponse(
        cache_key=result["cache_key"],
        synthetic_url=result["synthetic_url"],
        size_bytes=result["size_bytes"],
    )
```

```javascript
async function onUploadLocalClick() {
  const input = document.getElementById("local-file-input");
  const file = input.files && input.files[0];

  setExtractBusy(true);
  try {

    setStatus("Uploading local file…");

    await postInstructional(data.synthetic_url);
    await refreshCacheStatus();
    await refreshLibrary();
    await refreshCachePanel();
    setStatus("Uploaded — extract clips below.");

  } finally {
```
